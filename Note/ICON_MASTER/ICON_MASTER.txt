CREATE TABLE ICON_MASTER (
    ICON_ID           NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    ICON_NAME         VARCHAR2(100 BYTE) NOT NULL,         -- e.g., 'Home'
    ICON_CODE         VARCHAR2(200 BYTE) NOT NULL,         -- e.g., 'fa fa-home'
    ICON_LIBRARY      VARCHAR2(100 BYTE) NOT NULL,         -- e.g., 'FontAwesome'
    ICON_CATEGORY     VARCHAR2(100 BYTE),                  -- e.g., 'Navigation'
    PREVIEW_HTML      VARCHAR2(400 BYTE),
    ACTIVE            CHAR(1 BYTE) DEFAULT 'Y',
    CREATED_BY        VARCHAR2(100 BYTE),
    CREATED_ON        DATE DEFAULT SYSDATE,
MODIFIED_BY       VARCHAR2(100 BYTE),
    MODIFIED_DATE     DATE
);
